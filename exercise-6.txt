SELECT Phone.content AS "PhoneNumbers"
FROM Phone
JOIN Entry ON Phone.entryId = Entry.id
WHERE Entry.firstName = "Charlotte" AND Entry.lastName = "Jenkins"
GROUP BY content HAVING count(*) > 1;
